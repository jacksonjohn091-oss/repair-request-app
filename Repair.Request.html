    <script type="module">

        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";

        import { getAuth, signInAnonymously, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";

        import { getFirestore, doc, setDoc, getDoc, collection, onSnapshot, query, getDocs, updateDoc, deleteDoc } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";


        // Global variables provided by the canvas environment

        const firebaseConfig = JSON.parse(typeof __firebase_config !== 'undefined' ? __firebase_config : null);

        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';

        const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;

        

        // Initialize Firebase services

        const app = initializeApp(firebaseConfig);

        const db = getFirestore(app);

        const auth = getAuth(app);


        // Global variable to hold all requests

        let allRequests = [];


        // DOM element references

        const showFormButton = document.getElementById('show-form-button');

        const showSearchButton = document.getElementById('show-search-button');

        const showAllButton = document.getElementById('show-all-button');

        const exportButton = document.getElementById('export-button');

        const formContainer = document.getElementById('form-container');

        const searchContainer = document.getElementById('search-container');

        const allRequestsContainer = document.getElementById('all-requests-container');

        const allSections = document.querySelectorAll('.view-section');


        const repairForm = document.getElementById('repair-form');

        const formMessage = document.getElementById('form-message');


        const searchIdInput = document.getElementById('search-id');

        const searchButton = document.getElementById('search-button');

        const searchResults = document.getElementById('search-results');

        const searchMessage = document.getElementById('search-message');


        const allRequestsDiv = document.getElementById('all-requests');


        // Initial view setup

        showFormButton.addEventListener('click', () => {

            switchView('form-container');

        });

        showSearchButton.addEventListener('click', () => {

            switchView('search-container');

        });

        showAllButton.addEventListener('click', () => {

            switchView('all-requests-container');

        });

        exportButton.addEventListener('click', () => {

            exportToCsv();

        });


        /**

         * Switches the displayed view section.

         * @param {string} viewId The ID of the section to show.

         */

        function switchView(viewId) {

            allSections.forEach(section => {

                section.style.display = 'none';

            });

            const selectedSection = document.getElementById(viewId);

            if (selectedSection) {

                selectedSection.style.display = 'block';

            }

        }


        /**

         * Signs in the user using the provided custom token or anonymously.

         * This is necessary to use the database.

         */

        async function authenticateUser() {

            try {

                if (initialAuthToken) {

                    await signInWithCustomToken(auth, initialAuthToken);

                    console.log("Signed in with custom token.");

                } else {

                    await signInAnonymously(auth);

                    console.log("Signed in anonymously.");

                }

            } catch (error) {

                console.error("Authentication error:", error);

                showMessage(formMessage, 'Authentication failed. Please try again.', 'error');

            }

        }


        /**

         * Fetches all requests from Firestore and listens for real-time changes.

         * The data is stored in the 'allRequests' global variable.

         */

        const fetchAllRequests = () => {

            if (!auth.currentUser) {

                console.log("Authentication not ready. Skipping data fetch.");

                return;

            }


            const requestsRef = collection(db, `artifacts/${appId}/public/data/requests`);

            const q = query(requestsRef);


            // onSnapshot sets up a real-time listener

            onSnapshot(q, (snapshot) => {

                allRequests = [];

                snapshot.forEach((doc) => {

                    allRequests.push({ id: doc.id, ...doc.data() });

                });

                renderAllRequests();

            }, (error) => {

                console.error("Error fetching requests:", error);

            });

        };


        // Call authentication on page load

        auth.onAuthStateChanged(user => {

            if (user) {

                fetchAllRequests();

            } else {

                authenticateUser().then(() => {

                    fetchAllRequests();

                });

            }

        });


        // Form Submission Logic

        repairForm.addEventListener('submit', async (e) => {

            e.preventDefault();


            if (!auth.currentUser) {

                showMessage(formMessage, 'Please wait, authenticating with the server...', 'error');

                return;

            }


            const formData = new FormData(repairForm);

            const data = Object.fromEntries(formData.entries());


            // --- Start of new ID generation logic ---

            let newIdNumber = 1;

            try {

                // Fetch all documents to find the highest number

                const requestsRef = collection(db, `artifacts/${appId}/public/data/requests`);

                const querySnapshot = await getDocs(requestsRef);


                let maxId = 0;

                querySnapshot.forEach(doc => {

                    const idParts = doc.id.split('-');

                    if (idParts[0] === 'Req' && idParts.length > 1) {

                        const idNum = parseInt(idParts[1]);

                        if (!isNaN(idNum) && idNum > maxId) {

                            maxId = idNum;

                        }

                    }

                });

                newIdNumber = maxId + 1;

            } catch (error) {

                console.error("Error getting max ID, defaulting to 1:", error);

            }

            const docId = `Req-${newIdNumber}`;

            // --- End of new ID generation logic ---


            data.timestamp = new Date().toISOString();

            

            try {

                await setDoc(doc(db, `artifacts/${appId}/public/data/requests`, docId), data);


                showMessage(formMessage, `Success! Request submitted with ID: <span style="font-weight: bold;">${docId}</span>`, 'success');


                const copyButton = document.createElement('button');

                copyButton.className = 'copy-button';

                copyButton.textContent = 'Copy ID';

                formMessage.appendChild(copyButton);


                copyButton.addEventListener('click', () => {

                    const tempInput = document.createElement('input');

                    document.body.appendChild(tempInput);

                    tempInput.value = docId;

                    tempInput.select();

                    document.execCommand('copy');

                    document.body.removeChild(tempInput);

                    showMessage(formMessage, `Request ID Copied!`, 'success');

                });


                repairForm.reset();

            } catch (error) {

                console.error('Submission error:', error);

                showMessage(formMessage, `Failed to submit request: ${error.message}`, 'error');

            }

        });


        // Search Logic

        searchButton.addEventListener('click', async () => {

            const requestId = searchIdInput.value.trim();

            if (!requestId) {

                showMessage(searchMessage, 'Please enter a Request ID.', 'error');

                return;

            }


            if (!auth.currentUser) {

                showMessage(searchMessage, 'Please wait, authenticating with the server...', 'error');

                return;

            }


            searchResults.innerHTML = '<p>Searching...</p>';

            searchMessage.innerHTML = '';


            try {

                const docRef = doc(db, `artifacts/${appId}/public/data/requests`, requestId);

                const docSnap = await getDoc(docRef);


                if (docSnap.exists()) {

                    const foundRequest = docSnap.data();

                    renderEditableForm(foundRequest, requestId);

                    showMessage(searchMessage, 'Request found!', 'success');

                } else {

                    searchResults.innerHTML = '<p>No request found with that ID.</p>';

                    showMessage(searchMessage, 'No request found.', 'error');

                }

            } catch (error) {

                console.error('Search error:', error);

                showMessage(searchMessage, `An error occurred while searching: ${error.message}`, 'error');

            }

        });


        // Render all requests

        function renderAllRequests() {

            if (allRequests.length === 0) {

                allRequestsDiv.innerHTML = '<p>No requests found in the database.</p>';

                return;

            }


            allRequestsDiv.innerHTML = '';

            // Sort requests by ID number in descending order (latest first)

            const sortedRequests = allRequests.sort((a, b) => {

                const idA = parseInt(a.id.split('-')[1]);

                const idB = parseInt(b.id.split('-')[1]);

                return idB - idA;

            });

            

            sortedRequests.forEach(request => {

                const item = document.createElement('div');

                item.className = 'request-item';

                item.innerHTML = `

                    <h3>Request ID: ${request.id}</h3>

                    <p><strong>Name:</strong> ${request.name}</p>

                    <p><strong>Email:</strong> ${request.email}</p>

                    <p><strong>Date Requested:</strong> ${request.dateRequested}</p>

                    <p><strong>Reporter:</strong> ${request.reporterName || 'N/A'}</p>

                    <p><strong>Contact Person:</strong> ${request.contactPerson}</p>

                    <p><strong>Follow Up Notes:</strong> ${request.followUpNotes || 'N/A'}</p>

                `;

                allRequestsDiv.appendChild(item);

            });

        }


        /**

         * Exports all requests to a CSV file.

         */

        function exportToCsv() {

            if (allRequests.length === 0) {

                showMessage(searchMessage, 'No data to export.', 'error');

                return;

            }


            const headers = ['id', ...Object.keys(allRequests[0])];

            const csvRows = [];

            csvRows.push(headers.join(','));


            for (const request of allRequests) {

                const values = headers.map(header => {

                    const value = request[header];

                    const stringValue = (value || '').toString().replace(/"/g, '""');

                    return `"${stringValue}"`;

                });

                csvRows.push(values.join(','));

            }


            const csvString = csvRows.join('\n');

            const blob = new Blob([csvString], { type: 'text/csv;charset=utf-8;' });

            const link = document.createElement("a");

            const url = URL.createObjectURL(blob);

            link.setAttribute("href", url);

            link.setAttribute("download", "repair_requests.csv");

            link.style.visibility = 'hidden';

            document.body.appendChild(link);

            link.click();

            document.body.removeChild(link);

            showMessage(searchMessage, 'Exported all data to a CSV file.', 'success');

        }


        /**

         * Renders the editable form for a found request.

         * @param {object} data The request data to populate the form.

         * @param {string} docId The ID of the document to update.

         */

        function renderEditableForm(data, docId) {

            searchResults.innerHTML = `

                <h3>Edit Request: ${docId}</h3>
